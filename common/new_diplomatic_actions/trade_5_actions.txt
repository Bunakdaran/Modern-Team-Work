low_ivory_50 = {
	category = papacy
	require_acceptance = yes
	alert_index = 35
	alert_tooltip = ICON_TRADELEAGUEINVITE
	
is_visible = {
			has_country_flag = opened_trade_diplomacy 
        ai = no
		OR = {
				production_leader = {
						trade_goods = ivory
					}
				production_leader = {
						trade_goods = grain  
					}
				production_leader = {
						trade_goods = wine    
					}
				production_leader = {
						trade_goods = gold    
				}
			}
			FROM = { ivory = 1 }
	}
	
	is_allowed = {
			has_opinion_modifier = { 
				modifier = create_embassy  
				who = FROM 
			}		
		ROOT = {
			#prestige = 30
			stability = 1
			treasury = 7500
			total_development = 300
			
			
			NOT = { 	
				ROOT = { 
					has_country_modifier = failling_ivory_price_root_1
				}
			}
			NOT = { 	
				ROOT = { 
					has_country_modifier = rising_ivory_price_root_1
				}
			}
			OR = {
				production_leader = {
						trade_goods = ivory
					}
				production_leader = {
						trade_goods = grain  
					}
				production_leader = {
						trade_goods = wine    
					}
				production_leader = {
						trade_goods = gold    
				}
			}
			
		}
		FROM = {
			
			treasury = 6000
			stability = 1
			total_development = 300
			NOT = {
			FROM = { 
				has_country_modifier = failling_ivory_price_from
			}
		}
			ivory = 1

			
		}
		}
	
	
	on_accept = {
	
		change_price = {
			trade_goods = ivory
			key = IVORY_LOW
			value = -0.5
			duration = 3650
		}
	
		every_country = {
				
		
			if = {	
				limit = {
					NOT = { tag = ROOT }
					NOT = { tag = FROM }
					ivory = 1
				}
				add_country_modifier = {
							name = much_ivory_price_1_50
							duration = 3650
				} 
			}
			if = {
				limit = {
					NOT = { tag = ROOT }
					NOT = { tag = FROM }
				}
				add_country_modifier = {
					name = low_ivory_price_1_50
					duration = 3650
				}
			}
			
			}
		
		
		
		ROOT = {

			add_treasury = -7500
			add_country_modifier = {
				name = failling_ivory_price_root
				duration = 3650
			}
			add_country_modifier = {
				name = failling_ivory_price_root_1
				duration = 3650
			}
			
			
		}
		FROM = { 
			add_treasury = -6000
			add_country_modifier = {
				name = failling_ivory_price_from
				duration = 3650
			}
			add_country_modifier = {
				name = failling_ivory_price_root_1
				duration = 3650
			}
		}
		
				
	}
	
	ai_acceptance = {
	
		add_entry = {
			name = BASE_EMMBASY
			limit = { always = yes }
			export_to_variable = {
				variable_name = ai_value 
				value = 10
			}
			multiply_variable = { which = ai_value value = -1 }
		}
		
		add_entry = {
				name = RIVAL_WITH
				limit = { FROM = { is_rival = ROOT } }
				export_to_variable = {
					variable_name = ai_value 
					value = 100
				}
				multiply_variable = { which = ai_value value = -1 }
		}
		
		add_entry = {
			name = FRIENDLY_ATTITUDE
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_friendly } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
		}
		
		add_entry = {
			name = NEUTRAL_ATTITUDE
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_neutral } } }
			export_to_variable = {
				variable_name = ai_value 
				value = 10
			}
	
		}
		add_entry = {
				name = FRIENDLY_ATTITUDE
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_threatened } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 10
				}
	
			}
			add_entry = {
				name = OUR_SUBJECT
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_loyal } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
			}
			add_entry = {
				name = OUR_OVERLORD
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_overlord } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
			}
		add_entry = {
			name = WE_HAVE_GOOD_OPINON
			limit = { FROM = { has_opinion = { who = ROOT value = 120 }   } }
			export_to_variable = {
					variable_name = ai_value 
					value = 15
			}
	
		}
		add_entry = {
				name = OPINION

					variable_name = ai_value
					value = opinion
					who = FROM
					with = THIS
				}
				divide_variable = { which = ai_value value = 5 }

		add_entry = {

				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
		
		}
		
	}
	
	ai_will_do = {	
		
		FROM = { has_opinion = { who = ROOT value = 145 }   }
		
	}

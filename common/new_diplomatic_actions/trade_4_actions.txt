low_spices_50 = {
	category = papacy
	require_acceptance = yes
	alert_index = 35
	alert_tooltip = ICON_TRADELEAGUEINVITE
	
is_visible = {
			has_country_flag = opened_trade_diplomacy 
		ai = no
		OR = {
				production_leader = {
						trade_goods = spices
					}
				production_leader = {
						trade_goods = sugar  
					}
				production_leader = {
						trade_goods = salt    
					}
				production_leader = {
						trade_goods = gold    
				}
			}
			FROM = { spices = 1 }
	}
	
	is_allowed = {
			has_opinion_modifier = { 
				modifier = create_embassy  
				who = FROM 
			}		
		ROOT = {
			#prestige = 30
			stability = 1
			treasury = 7500
			total_development = 300
			
			
			NOT = { 	
				ROOT = { 
					has_country_modifier = failling_spices_price_root_1
				}
			}
			NOT = { 	
				ROOT = { 
					has_country_modifier = rising_spices_price_root_1
				}
			}
			OR = {
				production_leader = {
						trade_goods = spices
					}
				production_leader = {
						trade_goods = sugar  
					}
				production_leader = {
						trade_goods = salt    
					}
				production_leader = {
						trade_goods = gold    
				}
			}
			
		}
		FROM = {
			
			treasury = 6000
			stability = 1
			total_development = 300
			NOT = {
			FROM = { 
				has_country_modifier = failling_spices_price_from
			}
		}
			spices = 1

			
		}
		}
	
	
	on_accept = {
	
		change_price = {
			trade_goods = spices
			key = SPICES_LOW
			value = -0.5
			duration = 3650
		}
	
			every_country = {

		
				if = {	
					limit = {
						NOT = { tag = ROOT }
						NOT = { tag = FROM }
						spices = 1
					}
					add_country_modifier = {
								name = much_spices_price_1_50
								duration = 3650
					} 
				}
				if = {
					limit = {
						NOT = { tag = ROOT }
						NOT = { tag = FROM }
					}
					add_country_modifier = {
						name = low_spices_price_1_50
						duration = 3650
					}
				}
			
			}
		
		
		
		ROOT = {

			add_treasury = -7500
			add_country_modifier = {
				name = failling_spices_price_root
				duration = 3650
			}
			add_country_modifier = {
				name = failling_spices_price_root_1
				duration = 3650
			}
			
			
		}
		FROM = { 
			add_treasury = -6000
			add_country_modifier = {
				name = failling_spices_price_from
				duration = 3650
			}
			add_country_modifier = {
				name = failling_spices_price_root_1
				duration = 3650
			}
		}
		
				
	}
	
	ai_acceptance = {
	
		add_entry = {
			name = BASE_EMMBASY
			limit = { always = yes }
			export_to_variable = {
				variable_name = ai_value 
				value = 10
			}
			multiply_variable = { which = ai_value value = -1 }
		}
		
		add_entry = {
				name = RIVAL_WITH
				limit = { FROM = { is_rival = ROOT } }
				export_to_variable = {
					variable_name = ai_value 
					value = 100
				}
				multiply_variable = { which = ai_value value = -1 }
		}
		
		add_entry = {
			name = FRIENDLY_ATTITUDE
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_friendly } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
		}
		
		add_entry = {
			name = NEUTRAL_ATTITUDE
			limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_neutral } } }
			export_to_variable = {
				variable_name = ai_value 
				value = 10
			}
	
		}
		add_entry = {
				name = FRIENDLY_ATTITUDE
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_threatened } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 10
				}
	
			}
			add_entry = {
				name = OUR_SUBJECT
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_loyal } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
			}
			add_entry = {
				name = OUR_OVERLORD
				limit = { FROM = { ai_attitude = { who = ROOT attitude = attitude_overlord } } }
				export_to_variable = {
					variable_name = ai_value 
					value = 15
				}
	
			}
		add_entry = {
			name = WE_HAVE_GOOD_OPINON
			limit = { FROM = { has_opinion = { who = ROOT value = 120 }   } }
			export_to_variable = {
					variable_name = ai_value 
					value = 15
			}
	
		}
		add_entry = {
				name = OPINION

					variable_name = ai_value

					who = FROM
					with = THIS

					variable_name = ai_value 
					value = 15

	}

		FROM = { has_opinion = { who = ROOT value = 145 }   }

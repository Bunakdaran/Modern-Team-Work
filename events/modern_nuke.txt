country_event = {
	id = et_nuke.50
	title = et_nuke.50.t
	desc = et_nuke.50.d
	picture = MISSILE_RANGE_eventPicture

	is_triggered_only = yes

	immediate = {
		event_target:nuke_target_country = {
			every_owned_province = {
				limit = {
					OR = {
						sea_superregion = {
							heavy_ships_in_province = root
							#OR = {
							#	controlled_by = root
							#	controller = {
							#		is_subject_of = root
							#	}
							#}
							NOT = {
								province_distance = {
									who = prev
									distance = 50
								}
							}
						}
						any_province = {
							country_or_subject_holds = root
							has_building = railroad
							OR = {
								controlled_by = root
								controller = {
									is_subject_of = root
								}
							}
							NOT = {
								province_distance = {
									who = prev
									distance = 50
								}
							}
						}
					}
					is_city = yes
				}
				set_province_flag = distance_city 	
			}
		}
	}
	option = { #Major City
		name = et_nuke.1.a50
		trigger = { 
			check_variable = {
				which = missil_50
				value = 1
			}
		}
		add_country_modifier = {
			name = modifire_distance_50
			duration = -1
		}
		country_event = {
			id = et_nuke.1
		}
		
	}
	option = { #Cancel
		name = et_nuke.1.cansel
		event_target:nuke_target_country = {
			every_owned_province = {
				limit = { 
					has_province_flag = distance_city
				}
				clr_province_flag = distance_city
			}
		}
		clear_global_event_target = nuke_target_country
	}
}
